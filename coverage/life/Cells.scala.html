37 /** [Cells] is a [Cell] collection.
38   */
39 type Cells = Set[Cell]
40 
41 object Cells:
42   type Error  = String
43   type Result = Either[Error, Cells]
44 
45   private def fromTextEx(s: String): Cells =
46     val rows = s
47       .replace(" ", "")     // Remove spaces
48       .split("\n")          // Split lines
49       .filterNot(_.isBlank) // Remove blank lines
50       .map(_.trim)          // Remove end of lines
51       .toList
52     val rowLengths = rows.map(_.length)
53     require(s.length > 0, "Invalid empty string")
54     require(Set.from(rowLengths).sizeIs == 1, s"Inconsistent row lengths $rowLengths")
55     val liveCells = for
56       r <- 0 until rows.length
57       row = rows(r)
58       c <- 0 until row.length if row(c) != '.'
59     yield Cell(r, c)
60     liveCells.toSet
61 
62   /** Create a set of Cells from a multi-lined string representation. The first character of the string represents offset 0, 0.
63     * Spaces and blank lines are removed, '.' are empty cells, anything else is a live cell.
64     *
65     * @param s
66     *   The string representation.
67     * @returns
68     *   Either the [Cells] or an [Error].
69     */
70   def fromText(s: String): Result =
71     Try(fromTextEx(s)) match
72       case Success(cells) => Right[Error, Cells](cells)
73       case Failure(error) => Left[Error, Cells](error.getMessage)
74 
75   private def fromFilenameEx(filename: String): Cells =
76     val source = Source.fromFile(filename)(Codec.defaultCharsetCodec)
77     fromTextEx(source.getLines().mkString("\n"))
78 
79   /** Create a set of Cells from the content of a regular text file. The first character in the file is at offset 0, 0. Spaces and
80     * blank lines are removed, '.' are empty cells, anything else is a live cell.
81     *
82     * @param filename
83     *   The filename (and path) of the file cotaining the [Cells] description.
84     * @returns
85     *   Either the [Cells] or an [Error].
86     */
87   def fromFilename(filename: String): Result =
88     Try(fromFilenameEx(filename)) match
89       case Success(cells) => Right[Error, Cells](cells)
90       case Failure(error) => Left[Error, Cells](error.getMessage)

92 extension (cells: Cells)
93   def bounds: Bounds =
94     val head :: rest = cells.toList: @unchecked
95     val headBounds   = (head.row, head.row, head.column, head.column)
96 
97     def updateBounds(bounds: (Int, Int, Int, Int), cell: Cell) =
98       (
99         Math.min(bounds._1, cell.row),
100         Math.max(bounds._2, cell.row),
101         Math.min(bounds._3, cell.column),
102         Math.max(bounds._4, cell.column)
103       )
104 
105     val (minRow, maxRow, minColumn, maxColumn) = rest.foldLeft(headBounds)(updateBounds)
106     Bounds(minRow to maxRow, minColumn to maxColumn)
107 
108   def rotate(n: Int) =
109     val realN = if n >= 0 then n % 4 else (4 + n % 4) % 4
110     val bounds = cells.bounds
111     val source = for
112       sourceRow    <- bounds.rows
113       sourceColumn <- bounds.columns
114     yield Cell(sourceRow, sourceColumn)
115 
116     val targets = Array(
117       source,
118       for (targetColumn <- bounds.rows.reverse; targetRow <- bounds.columns) yield Cell(targetRow, targetColumn),
119       for (targetRow <- bounds.rows.reverse; targetColumn <- bounds.columns.reverse) yield Cell(targetRow, targetColumn),
120       for (targetColumn <- bounds.rows; targetRow <- bounds.columns.reverse) yield Cell(targetRow, targetColumn)
121     )
122 
123     val mapping = source.zip(targets(realN)).toMap
124     cells.map(mapping)
