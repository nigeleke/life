33 /** A [World] on which the [Game of Life](https://en.wikipedia.org/wiki/Conway%27s_Game_of_Life) will be played out.
34   * @param liveCells
35   *   The population of live [Cell]s in the current [World].
36   * @param maybeBounds
37   *   The optional [Bounds] placed on the current [World]. If provided, the edges of the [World] are void, and any [Cell]s 'born'
38   *   into the edge immediately fall off the edge. If not provided then the [Bounds] are endless.
39   * @param maybeViewport
40   *   The optional [Bounds] for the view of the [World] when displaying. If provided, the [World] is only displayed within the range
41   *   of the viewport's rows and columns. If not provided then the full [World] is displayed.
42   */
43 final case class World(liveCells: Cells, maybeBounds: Option[Bounds], maybeViewport: Option[Bounds])
44 
45 object World:
46   /** Create a [World] with an initial population of [Cell]s. The [World] will have no boundaries, or restricted viewing.
47     * @constructor
48     * @param cells
49     *   The initial population.
50     * @return
51     *   The created [World].
52     */
53   def withCells(cells: Cells) = World(liveCells = cells, None, None)
54 
55 extension (w: World)
56   private def removingOffworlders: World =
57     def withinRange(r: Range, i: Int)    = r.contains(i)
58     def withinBounds(b: Bounds)(c: Cell) = withinRange(b.rows, c.row) && withinRange(b.columns, c.column)
59     def cellsInBounds(b: Bounds)         = w.liveCells.filter(withinBounds(b))
60     w.maybeBounds.map(bounds => w.copy(liveCells = cellsInBounds(bounds))).getOrElse(w)
61 
62   /** Restrict the boundaries of the current [World]. Any existing [Cell]s outside those boundaries will fall off the edge of the
63     * [World].
64     * @param bounds
65     *   The new bounds.
66     * @return
67     *   The [World] with the updated boundary, and cells removed.
68     */
69   def withBounds(bounds: Bounds): World = w.copy(maybeBounds = Some(bounds)).removingOffworlders
70 
71   /** Remove the [World]'s boundaries.
72     * @return
73     */
74   def unbounded: World = w.copy(maybeBounds = None)
75 
76   /** Set the current viewport for displaying.
77     * @param viewport
78     *   The new viewport.
79     * @return
80     *   The updated [World] with the new viewport.
81     */
82   def withViewport(viewport: Bounds): World = w.copy(maybeViewport = Some(viewport))
83 
84   /** Clear any pre-defined viewport. The whole [World] will be viewed when displaying.
85     * @return
86     *   The updated [World] with any existing viewport removed.
87     */
88   def withoutViewport: World = w.copy(maybeViewport = None)
89 
90   /** Add some additional [Cells] to the current [World].
91     * @param cells
92     *   The [Cells] to be added.
93     * @param row
94     *   The row offset in the current [World] where the new [Cells] will be added.
95     * @param column
96     *   The column offset in the current [World] where the new [Cells] will be added.
97     * @return
98     *   The [World] with the additional [Cells].
99     */
100   def withCells(cells: Cells, row: Int, column: Int): World =
101     val offsetCells = cells.map(c => Cell(c.row + row, c.column + column))
102     w.copy(liveCells = w.liveCells ++ offsetCells).removingOffworlders
103 
104   private def isLive(c: Cell): Boolean = w.liveCells.contains(c)
105 
106   private def neighboursCount(c: Cell): Int = c.neighbouringCells.count(isLive)
107 
108   /** @return
109     *   The next generation of the [World], using the rules of life.
110     */
111   def nextGeneration: World =
112     val allNeighbours   = w.liveCells.map(_.neighbouringCells).flatten
113     val allCells        = w.liveCells ++ allNeighbours
114     val neighbourCounts = allCells.map(c => (c, w.neighboursCount(c)))
115     val dyingCells      = neighbourCounts.filter((_, count) => count < 2 || count > 3).map(_._1)
116     val bornCells       = neighbourCounts.filter((_, count) => count == 3).map(_._1)
117     w.copy(liveCells = w.liveCells -- dyingCells ++ bornCells).removingOffworlders
118 
119   /** @return
120     *   A pretty string to aid debug & testing primarily.
121     */
122   def prettified: String =
123     val bounds = w.maybeViewport.getOrElse(w.maybeBounds.getOrElse(w.liveCells.bounds))
124     def rowString(r: Int): String =
125       (for
126         c <- bounds.columns
127         cellChar = if w.isLive(Cell(r, c)) then "* " else "  "
128       yield cellChar).mkString
129     (for r <- bounds.rows
130     yield rowString(r)).mkString("\n")
