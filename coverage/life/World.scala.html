33 /** A [World] on which the [Game of Life](https://en.wikipedia.org/wiki/Conway%27s_Game_of_Life) will be played out.
34   * @param liveCells
35   *   The population of live [Cell]s in the current [World].
36   * @param maybeBounds
37   *   The optional [Bounds] placed on the current [World]. If provided, the edges of the [World] are void, and any [Cell]s 'born'
38   *   into the edge immediately fall off the edge. If not provided then the [Bounds] are endless.
39   * @param maybeViewport
40   *   The optional [Bounds] for the view of the [World] when displaying. If provided, the [World] is only displayed within the range
41   *   of the viewport's rows and columns. If not provided then the full [World] is displayed.
42   */
43 final case class World(liveCells: Cells, maybeBounds: Option[Bounds], maybeViewport: Option[Bounds])
44 
45 object World:
46   /** Create a [World] with an initial population of [Cell]s. The [World] will have no boundaries, or restricted viewing.
47     * @constructor
48     * @param cells
49     *   The initial population.
50     * @return
51     *   The created [World].
52     */
53   def withCells(cells: Cells) = World(liveCells = cells, None, None)
54 
55 extension (w: World)
56   private def removingOffworlders: World =
57     def withinRange(r: Range, i: Int)    = r.contains(i)
58     def withinBounds(b: Bounds)(c: Cell) = withinRange(b.rows, c.row) && withinRange(b.columns, c.column)
59     def cellsInBounds(b: Bounds)         = w.liveCells.filter(withinBounds(b))
60     w.maybeBounds.map(bounds => w.copy(liveCells = cellsInBounds(bounds))).getOrElse(w)
61 
62   /** Restrict the boundaries of the current [World]. Any existing [Cell]s outside those boundaries will fall off the edge of the
63     * [World].
64     * @param bounds
65     *   The new bounds.
66     * @return
67     *   The [World] with the updated boundary, and cells removed.
68     */
69   def withBounds(bounds: Bounds): World = w.copy(maybeBounds = Some(bounds)).removingOffworlders
70 
71   /** Remove the [World]'s boundaries.
72     * @return
73     */
74   def unbounded: World = w.copy(maybeBounds = None)
75 
76   /** @return
77     *   The [World]'s boundaries. If defined, then that definition will be return. If the [World] is unbounded then the limits of
78     *   the existing cells are returned.
79     */
80   def bounds =
81     lazy val calculatedBounds =
82       val head :: rest = w.liveCells.toList: @unchecked
83       val headBounds   = (head.row, head.row, head.column, head.column)
84       def updateBounds(bounds: (Int, Int, Int, Int), cell: Cell) =
85         (
86           Math.min(bounds._1, cell.row),
87           Math.max(bounds._2, cell.row),
88           Math.min(bounds._3, cell.column),
89           Math.max(bounds._4, cell.column)
90         )
91       val (minRow, maxRow, minColumn, maxColumn) = rest.foldLeft(headBounds)(updateBounds)
92       Bounds(minRow to maxRow, minColumn to maxColumn)
93     w.maybeBounds.getOrElse(calculatedBounds)
94 
95   /** Set the current viewport for displaying.
96     * @param viewport
97     *   The new viewport.
98     * @return
99     *   The updated [World] with the new viewport.
100     */
101   def withViewport(viewport: Bounds): World = w.copy(maybeViewport = Some(viewport))
102 
103   /** Clear any pre-defined viewport. The whole [World] will be viewed when displaying.
104     * @return
105     *   The updated [World] with any existing viewport removed.
106     */
107   def withoutViewport: World = w.copy(maybeViewport = None)
108 
109   /** Add some additional [Cells] to the current [World].
110     * @param cells
111     *   The [Cells] to be added.
112     * @param row
113     *   The row offset in the current [World] where the new [Cells] will be added.
114     * @param column
115     *   The column offset in the current [World] where the new [Cells] will be added.
116     * @return
117     *   The [World] with the additional [Cells].
118     */
119   def withCells(cells: Cells, row: Int, column: Int): World =
120     val offsetCells = cells.map(c => Cell(c.row + row, c.column + column))
121     w.copy(liveCells = w.liveCells ++ offsetCells).removingOffworlders
122 
123   private def isLive(c: Cell): Boolean = w.liveCells.contains(c)
124 
125   private def neighboursCount(c: Cell): Int = c.neighbouringCells.count(isLive)
126 
127   /** @return
128     *   The next generation of the [World], using the rules of life.
129     */
130   def nextGeneration: World =
131     val allNeighbours   = w.liveCells.map(_.neighbouringCells).flatten
132     val allCells        = w.liveCells ++ allNeighbours
133     val neighbourCounts = allCells.map(c => (c, w.neighboursCount(c)))
134     val dyingCells      = neighbourCounts.filter((_, count) => count < 2 || count > 3).map(_._1)
135     val bornCells       = neighbourCounts.filter((_, count) => count == 3).map(_._1)
136     w.copy(liveCells = w.liveCells -- dyingCells ++ bornCells).removingOffworlders
137 
138   /** @return
139     *   A pretty string to aid debug & testing primarily.
140     */
141   def prettified: String =
142     val bounds = w.maybeViewport.getOrElse(w.bounds)
143     def rowString(r: Int): String =
144       (for
145         c <- bounds.columns
146         cellChar = if w.isLive(Cell(r, c)) then "* " else "  "
147       yield cellChar).mkString
148     (for r <- bounds.rows
149     yield rowString(r)).mkString("\n")
