2
3use std::cmp::{max, min};
4use std::ops::RangeInclusive;
5
6#[derive(Clone, Debug, Default, PartialEq)]
7pub enum Bounds {
8    #[default]
9    Undefined,
10    Defined(RangeInclusive<isize>, RangeInclusive<isize>),
11}
12
13impl Bounds {
14    pub fn encompass(&mut self, cell: &Cell) {
15        *self = match self {
16            Bounds::Undefined => {
17                Bounds::Defined(cell.row()..=cell.row(), cell.column()..=cell.column())
18            }
19            Bounds::Defined(rows, columns) => {
20                let (r_min, r_max) = (*rows.start(), *rows.end());
21                let (c_min, c_max) = (*columns.start(), *columns.end());
22                let rows = min(r_min, cell.row())..=max(r_max, cell.row());
23                let columns = min(c_min, cell.column())..=max(c_max, cell.column());
24                Bounds::Defined(rows, columns)
25            }
26        };
27    }
28
29    pub fn rows(&self) -> Option<&RangeInclusive<isize>> {
30        match self {
31            Bounds::Undefined => None,
32            Bounds::Defined(rows, _) => Some(&rows),
33        }
34    }
35
36    pub fn columns(&self) -> Option<&RangeInclusive<isize>> {
37        match self {
38            Bounds::Undefined => None,
39            Bounds::Defined(_, columns) => Some(&columns),
40        }
41    }
42}
