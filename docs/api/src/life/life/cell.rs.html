9impl Cell {

16    }

22    pub fn neighbours(&self) -> Vec<Cell> {
23        (-1..=1)
24            .flat_map(|delta_row| {
25                (-1..=1)
26                    .filter(move |delta_column| (delta_row != 0 || *delta_column != 0))
27                    .map(move |delta_column| *self + Position::new(delta_row, delta_column))
28            })
29            .collect::<Vec<_>>()
30    }
31}
