6fn parse_range(s: &str) -> Result<RangeInclusive<isize>, String> {
7    let min_max = s.split("..").collect::<Vec<_>>();
8    if min_max.len() != 2 {
9        return Err("bounds format rowmin..rowmax,colmin..colmax".into());
10    }
11    let min = min_max[0].parse::<isize>().map_err(|e| e.to_string())?;
12    let max = min_max[1].parse::<isize>().map_err(|e| e.to_string())?;
13    Ok(min..=max)
14}
15
16fn parse_bounds(s: &str) -> Result<Bounds, String> {
17    let ranges = s.split(",").collect::<Vec<_>>();
18    if ranges.len() != 2 {
19        return Err("bounds format rowmin..rowmax,colmin..colmax".into());
20    };
21
22    let rows = parse_range(ranges[0])?;
23    let columns = parse_range(ranges[1])?;
24    Ok(Bounds::Defined(rows, columns))
25}
