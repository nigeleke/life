40 object Life:
41 
42   val config: Config = ConfigFactory.load().getConfig("life")
43   val board          = Bounds.from(config.getConfig("defaults.boardBounds"))
44   val viewport       = Bounds.from(config.getConfig("defaults.viewportBounds"))
45 
46   private def iterateGenerations(cells: Cells): Unit =
47     val initialWorld = World.withCells(cells).withBounds(board).withViewport(viewport)
48     print(Ansi.clearScreen)
49     Generations(initialWorld).foreach { world =>
50       print(Ansi.home)
51       println(world.prettified)
52       Thread.sleep(200)
53     }
54 
55   private def runRandom: Unit =
56     val cells = Patterns.random(board)
57     iterateGenerations(cells)
58 
59   private def runWithFile(filename: String): Unit =
60     Cells.fromFilename(filename) match
61       case Right(cells) => iterateGenerations(cells)
62       case Left(error)  => println(error)
63 
64   def main(args: Array[String]): Unit =
65     if args.isEmpty
66     then runRandom
67     else args.foreach(runWithFile)

72     val home              = s"$esc[H"
