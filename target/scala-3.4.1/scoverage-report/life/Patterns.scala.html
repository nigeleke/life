114   /** Randomize a board's population. Cells are created in approxiamately 20% of the given board.
115     * @param board
116     *   The [Bounds] of the board to be populated.
117     * @return
118     *   The randomized population.
119     */
120   def random(board: Bounds): Cells =
121     (for
122       row    <- board.rows
123       column <- board.columns
124       if Random.nextDouble <= 0.2
125     yield Cell(row, column)).toSet
